NONSTOP FLIGHT SMALLEST TIME PER DATE AND AIRPORT -- 
select a2.date, a2.flightnumber, a2.airlinecode, min(a2.diff) as smallest from 
	(select date, airlinecode, flightnumber, case 
		when a1.diff < '00:00' then a1.diff + '24:00'
		else a1.diff
		END
		from (select date, airlinecode, flightnumber, dtime - time as diff 
				from flight where iata ='ATL' and diata = 'ORD' and date = '2017-12-12') as a1) as a2
				group by airlinecode, flightnumber, date order by smallest

ONE CONNECTION -- select * from 
			flight as a1 join flight on a1.diata = flight.iata
			where a1.iata ='JFK' 
			and flight.diata = 'DFW' 
			and a1.ddate < flight.date 
			and a1.dtime < flight.time

TWO CONNECTIONS -- select * from 
			flight as a1 join flight as a2 on a1.diata = a2.iata join flight on a2.diata = flight.iata
			where a1.iata = 'ORD' 
			and flight.diata = 'DFW' 
			and a1.ddate <= a2.date 
			and a2.ddate <= flight.date 
			and a1.dtime < a2.time 
			and a2.dtime < flight.time
